    <h2>Bulk Find & Replace Tool</h2>

    <p>Enter your main text:</p>

    <textarea id="text" rows="6" placeholder="Paste your text here..."></textarea>


    <p>Enter Find values (one per line):</p>

    <textarea id="find" rows="6" placeholder="One value per line"></textarea>


    <p>Enter Replace values (matching order):</p>

    <textarea id="replace" rows="6" placeholder="One value per line"></textarea>


    <button onclick="process()">Replace</button>


    <h3>Output:</h3>

    <div id="output"></div>


    <p style="font-size: 12px; margin-top: 20px;">

      Like this tool? Support at <a href="https://www.buymeacoffee.com/donate2mrpress" target="_blank" rel="noopener">buymeacoffee.com/donate2mrpress</a>

    <script>

      function process() {

        const text = document.getElementById('text').value;

        const find = document.getElementById('find').value;

        const replace = document.getElementById('replace').value;


        const finds = find.split('\n');

        const replaces = replace.split('\n');


        if (finds.length !== replaces.length) {

          document.getElementById('output').textContent = 'Error: Find and Replace lists must be the same length.';

          return;

        }


        let result = text;

        for (let i = 0; i < finds.length; i++) {

          const findVal = finds[i];

          const replaceVal = replaces[i];

          const regex = new RegExp(findVal, 'g');

          result = result.replace(regex, replaceVal);

        }


        document.getElementById('output').textContent = result;

      }

    </script>
